# 数据库设计文档 V1

# 基于 TurboPush 改造的 AI 内容运营中台

## 一、设计目标

本数据库设计服务于第一阶段 MVP，目标是支撑以下核心流程：

```
选题创建
→ IMA 搜索参考
→ Agent 生成内容
→ 多平台版本生成
→ 人工审核
→ TurboPush 发布
→ 平台数据回收
→ 数据复盘
```

当前技术约束：

- 数据库：SQLite
- ORM：Prisma
- 不使用 Redis
- 不使用对象存储
- 不使用向量数据库
- 不自研知识库
- IMA 只作为外部搜索能力
- TurboPush 作为账号、发布和平台适配底座

## 六、表关系说明

| 子表 | 外键字段 | 父表 | 关系 | 删除策略 |
| ---- | -------- | ---- | ---- | -------- |
| account_profiles | accountId | platform_accounts | 1:1 | Cascade |
| topics | ownerId | users | N:1 | Restrict |
| contents | topicId | topics | N:1 | SetNull |
| contents | createdBy | users | N:1 | Restrict |
| content_versions | contentId | contents | N:1 | Cascade |
| content_versions | accountId | platform_accounts | N:1 | Restrict |
| materials | contentId | contents | N:1 | Cascade |
| agents | promptId | prompts | N:1 | Restrict |
| agent_runs | agentId | agents | N:1 | Restrict |
| agent_runs | contentId | contents | N:1 | Cascade |
| agent_runs | versionId | content_versions | N:1 | SetNull |
| ima_search_logs | contentId | contents | N:1 | Cascade |
| review_tasks | contentId | contents | N:1 | Cascade |
| review_tasks | versionId | content_versions | N:1 | SetNull |
| review_tasks | reviewerId | users | N:1 | SetNull |
| publishing_tasks | contentId | contents | N:1 | Cascade |
| publishing_tasks | versionId | content_versions | N:1 | Restrict |
| publishing_tasks | accountId | platform_accounts | N:1 | Restrict |
| publish_records | taskId | publishing_tasks | 1:1 | Cascade |
| analytics_data | publishRecordId | publish_records | N:1 | Cascade |
| analytics_reports | contentId | contents | N:1 | Cascade |
| platform_accounts | ownerId | users | N:1 | Restrict |

说明：

- `platform_accounts` 与 TurboPush 原表映射时，优先保持 `id` 一致，便于 Adapter 同步。
- 内容主表删除时，级联删除版本、素材、Agent 执行记录等从属数据。
- 审核人、版本 ID 可为空，支持「仅内容级审核」场景。

## 八、Prisma Schema

以下为可直接放入 `packages/db/prisma/schema.prisma` 的第一版 Schema（与上文字段、枚举一致）。

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = env("DATABASE_URL")
}

enum UserRole {
  ADMIN
  OPERATOR
  REVIEWER
  ANALYST
}

enum Platform {
  WECHAT
  XIAOHONGSHU
  DOUYIN
  VIDEO_CHANNEL
  BILIBILI
  ZHIHU
  OTHER
}

enum ContentStatus {
  DRAFT
  PENDING_GENERATE
  GENERATING
  PENDING_REVIEW
  REJECTED
  APPROVED
  PENDING_PUBLISH
  PUBLISHING
  PUBLISHED
  FAILED
  REVIEWED
  ARCHIVED
}

enum AgentType {
  TITLE
  TAG
  REWRITE
  BODY
  COVER_COPY
  REVIEW
  SUMMARY
  TOPIC
  IMAGE
  VIDEO_SCRIPT
  COMPETITOR
}

enum RunStatus {
  PENDING
  RUNNING
  SUCCESS
  FAILED
}

enum ReviewStatus {
  PENDING
  APPROVED
  REJECTED
}

enum PublishStatus {
  PENDING
  PUBLISHING
  SUCCESS
  FAILED
  CANCELLED
}

model User {
  id        String   @id @default(cuid())
  name      String
  email     String   @unique
  role      UserRole @default(OPERATOR)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  topics          Topic[]
  contents        Content[]         @relation("ContentCreator")
  reviewTasks     ReviewTask[]      @relation("Reviewer")
  ownedAccounts   PlatformAccount[] @relation("AccountOwner")

  @@map("users")
}

model PlatformAccount {
  id          String   @id @default(cuid())
  platform    Platform
  accountName String
  accountType String?
  authStatus  String   @default("pending")
  ownerId     String
  rawData     Json?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  owner           User              @relation("AccountOwner", fields: [ownerId], references: [id])
  profile         AccountProfile?
  contentVersions ContentVersion[]
  publishingTasks PublishingTask[]
  analyticsData   AnalyticsData[]

  @@index([platform, authStatus])
  @@index([ownerId])
  @@map("platform_accounts")
}

model AccountProfile {
  id              String   @id @default(cuid())
  accountId       String   @unique
  positioning     String?
  targetAudience  Json?
  contentStyle    String?
  titlePreference String?
  coverPreference String?
  tone            String?
  forbiddenWords  Json?
  contentBoundary String?
  publishStrategy String?
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  account PlatformAccount @relation(fields: [accountId], references: [id], onDelete: Cascade)

  @@map("account_profiles")
}

model Topic {
  id              String        @id @default(cuid())
  title           String
  description     String?
  source          String        @default("manual")
  targetPlatforms Json?
  status          ContentStatus @default(DRAFT)
  ownerId         String
  createdAt       DateTime      @default(now())
  updatedAt       DateTime      @updatedAt

  owner    User      @relation(fields: [ownerId], references: [id])
  contents Content[]

  @@index([ownerId, status, createdAt])
  @@map("topics")
}

model Content {
  id          String        @id @default(cuid())
  topicId     String?
  title       String
  summary     String?
  body        String?
  coverText   String?
  status      ContentStatus @default(DRAFT)
  aiGenerated Boolean       @default(false)
  createdBy   String
  createdAt   DateTime      @default(now())
  updatedAt   DateTime      @updatedAt

  topic           Topic?            @relation(fields: [topicId], references: [id], onDelete: SetNull)
  creator         User              @relation("ContentCreator", fields: [createdBy], references: [id])
  versions        ContentVersion[]
  materials       Material[]
  agentRuns       AgentRun[]
  imaSearchLogs   ImaSearchLog[]
  reviewTasks     ReviewTask[]
  publishingTasks PublishingTask[]
  analyticsReports AnalyticsReport[]
  analyticsData    AnalyticsData[]

  @@index([topicId, status, createdAt])
  @@index([createdBy])
  @@map("contents")
}

model ContentVersion {
  id           String        @id @default(cuid())
  contentId    String
  platform     Platform
  accountId    String?
  title        String?
  body         String?
  coverText    String?
  tags         Json?
  formatConfig Json?
  status       ContentStatus @default(DRAFT)
  createdAt    DateTime      @default(now())
  updatedAt    DateTime      @updatedAt

  content         Content          @relation(fields: [contentId], references: [id], onDelete: Cascade)
  account         PlatformAccount? @relation(fields: [accountId], references: [id])
  agentRuns       AgentRun[]
  reviewTasks     ReviewTask[]
  publishingTasks PublishingTask[]
  analyticsData   AnalyticsData[]

  @@index([contentId, platform])
  @@index([accountId, status])
  @@map("content_versions")
}

model Material {
  id        String   @id @default(cuid())
  contentId String
  type      String
  name      String?
  url       String?
  localPath String?
  source    String?
  meta      Json?
  createdAt DateTime @default(now())

  content Content @relation(fields: [contentId], references: [id], onDelete: Cascade)

  @@index([contentId])
  @@map("materials")
}

model Prompt {
  id        String    @id @default(cuid())
  name      String
  agentType AgentType
  version   String    @default("v1")
  template  String
  variables Json?
  enabled   Boolean   @default(true)
  createdAt DateTime  @default(now())
  updatedAt DateTime  @updatedAt

  agents Agent[]

  @@index([agentType, enabled])
  @@map("prompts")
}

model Agent {
  id          String    @id @default(cuid())
  name        String
  type        AgentType
  description String?
  promptId    String
  model       String    @default("deepseek-chat")
  enabled     Boolean   @default(true)
  config      Json?
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt

  prompt    Prompt     @relation(fields: [promptId], references: [id])
  agentRuns AgentRun[]

  @@index([type, enabled])
  @@map("agents")
}

model AgentRun {
  id            String    @id @default(cuid())
  agentId       String
  contentId     String
  versionId     String?
  input         Json?
  output        Json?
  model         String?
  promptVersion String?
```
